[docs]def hex_to_rgb(hex, ansi = True):
    stripped_hex = hex.lstrip('#')
    r = int(stripped_hex[0:2], 16)
    g = int(stripped_hex[2:4], 16)
    b = int(stripped_hex[4:6], 16)
    if ansi:
        return f'{r};{g};{b}'
    else:
        return tuple(r, g, b)



[docs]def rgb_to_hex(rgb):
    return f'#{rgb[0]:x}{rgb[1]:x}{rgb[2]:x}'.upper()
